class Solution {
public:
    int distributeCandies(vector<int>& candyType) {
        unordered_set<int> types;

        for(int i = 0; i < candyType.size(); i++) {
            if(types.count(candyType[i]) == 0) {
                types.insert(candyType[i]);
            }
        }
        return min(types.size(), candyType.size()/2); 
    }
};
